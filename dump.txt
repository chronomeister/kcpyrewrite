/* プレースホルダー用 JavaScript */
(function() {
    var placeholderJS = (function() {
        var inputPlaceholderFlag = document.createElement('input').placeholder;

        /**
         * placeholderAction
         * placeholderの表示・非表示アクション
         */
        var placeholderAction = {
            /**
             * hide
             * thisの参照先をinputにしてください
             */
            hide: function() {
                $(this).nextAll(".placeholder").hide();
            },
            /**
             * show
             * thisの参照先をinputにしてください
             */
            show: function() {
                if ($(this).val() === "") {
                    $(this).nextAll(".placeholder").show();
                }
            }
        }

        /**
         * createPlaceholderElement
         *
         * @param {jQueryObject} $inputElement
         * @param {string} str placeholder文字列
         * @return {jQueryObject} $placeholderElement
         */
        function createPlaceholderElement($inputElement, str) {
            var $placeholder = $('<span>', {
                'class': 'placeholder'
            }).append(str);

            $inputElement.after($placeholder);

            return $placeholder;
        }

        /**
         * addEventPlaceholder
         * placeholderにクリックイベントを付与する
         *
         * @param {jQueryObject} $placeholderElement
         */
        function addEventPlaceholder($placeholderElement) {
            // placeholderはdisplay: none状態にする
            // DOMContentLoaded後 input valueの値を取得し
            // なにも表示されていなかった時にdisplay: noneを解除する
            $placeholderElement.hide().click(function () {
                $(this).prevAll("input").focus();
            });
        }

        /**
         * addEventInput
         * inputにイベントを付与する
         *
         * @param {jQueryObject} $inputElement
         */
        function addEventInput($inputElement) {
            $inputElement.focus(function () {
                placeholderAction.hide.apply(this);
            }).blur(function () {
                placeholderAction.show.apply(this);
            }).change(function () {
                placeholderAction.show.apply(this);
            }).on('DOMSubtreeModified propertychange', function() {
                placeholderAction.hide.apply(this);
            });

            // 初期状態のplaceholderは display: none状態なので
            // 何も入力されていなかった場合のみ表示させる
            $(function() {
                var val = $inputElement.val();

                if(val === '') {
                    placeholderAction.show.apply($inputElement[0]);
                }
            });
        }

        return {
            /**
             * createElement
             * placeholderを作成します input placeholderに対応している場合は実行を終了する
             * @param  {jQueryObject} $inputElement placeholderを挿入するinput
             * @param  {string} str
             */
            createElement: function($inputElement, str) {
                // input placeholderに対応していない場合のみ createPlaceholderElementを実行する
                if(inputPlaceholderFlag !== undefined) {
                    return;
                }

                var $placeholderElement = createPlaceholderElement($inputElement, str);
                addEventPlaceholder($placeholderElement);
                addEventInput($inputElement);
            }
        };
    })();

function createFooterBanner(ad)
{
    var img = document.createElement("img");
    var link = document.createElement("a");
    link.href = ad.clickUrl;

    if (ad.creative.images && ad.creative.images[0]) {
        //コンボクリエイティブの場合
        if (ad.creative.content) {
            for (contentIndex in ad.creative.content) {
                //オプションが設定されている場合
                switch (ad.creative.content[contentIndex].key) {
                case "target":
                    //管理画面でtargetの値を登録しないとcontent属性値が空配列になることに注意！
                    link.target = ad.creative.content[contentIndex].value;
                    break;
                case "説明":
                    //説明が登録されていたら代替テキストをセット
                    img.alt = ad.creative.content[contentIndex].value;
                    break;
                default:
                }
            }
        }

        img.src = ad.creative.images[0].source;
    } else {
        //イメージクリエイティブの場合
        img.src = ad.creative.source;

        if (ad.creative.description) {
            //説明が登録されていたら代替テキストをセット
            img.alt = ad.creative.description;
        }
    }

    link.appendChild(img);
    return link;
}

//広告配信サーバへのリクエスト送信に対するコールバック関数
function renderFooterAd(data, context) {
    if (isFooterAd) return;
    if (data && data.searchResult && data.searchResult.spaces && data.searchResult.spaces[0] && data.searchResult.spaces[0].ads) {
        //広告スペースが存在する場合は広告オブジェクトを取得
        var ads = data.searchResult.spaces[0].ads;
        if (ads[0]) {
            //フッターバナー
            document.querySelector("#footer-ad-container").appendChild(createFooterBanner(ads[0]));
            isFooterAd = true;
        }
    } else {
        //登録バナーが足りないときやアドサーバ側のバグによって意図する広告を取得できなかった場合
        insertAlternateFooterAd();
    }
}

            $("#footer-ad-container").append(adElement);
            isFooterAd = true;
        }
    } catch (e) {
        //何もしない
    }
}
